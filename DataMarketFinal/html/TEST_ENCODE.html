## The function is currently defined as
function (data) 
{
    n = dim(data)[1]
    if (n > 0) {
        guess1 = tryCatch(nchar(data[1, ], keepNA = F), warning = function(msg) return("W"), 
            error = function(e) return("e"))
    }
    else {
        guess1 = 0
    }
    if (n > 4) {
        guess2 = tryCatch(nchar(data[5, ], keepNA = F), warning = function(msg) return("W"), 
            error = function(e) return("e"))
    }
    else {
        guess2 = 0
    }
    guess3 = tryCatch(nchar(colnames(data), keepNA = F), warning = function(msg) return("W"), 
        error = function(e) return("e"))
    if (class(c(guess1, guess2, guess3)) == "character") {
        return("Maybe wrong encode")
    }
    else {
        return("OK")
    }
  }
