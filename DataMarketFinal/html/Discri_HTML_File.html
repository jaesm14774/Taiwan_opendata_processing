Examples

writeLines(text=test_data2,con=tmp)
Discri_HTML_File(data_path=tmp)
unlink(tmp)

## The function is currently defined as
function (file_path) 
{
    yes = "<html|<style|font-size|<span|body  onload|background-|<script|color|\.style|<meta|<open|<head|<!DOCTYPE|<title|<body|<kml|<name|<SimpleField|<link|<div|<table|<input|<form|</"
    code = SEARCH_ENCODE(file_path)
    discri_text = readLines(con = file_path, n = 25, encoding = ifelse(code == 
        "unknown", "UTF-8", code))
    ifelse(sum(grepl(discri_text, pattern = yes)) > 4, TRUE, 
        FALSE) %>% return
  }
